{

	int n;

	clrscr();

	printf("Factorial of %d is %d", n, factorial(n));
	getch();

}

long int factorial(int n)

{

	if (n <= 1)

	{

		return (1);

	}

	else

	{

		n = n * factorial(n - 1);
		return (n);

	}
}
