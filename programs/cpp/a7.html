Inverse Matrix

/*Inverse Matrix using Gauss jordan method*/

#include<iostream.h>
#include<conio.h>

void main()
{
    int i,j,k,n;
    float a[10][10]={0},d;

    clrscr();
    cout<<"Enter the order of matrix : ";
    cin>>n;
    cout<<"Enter the elements : "<<endl;
    for(i=1;i<=n;i++)
        for(j=1;j<=n;j++)
            cin>>a[i][j];

    for(i=1;i<=n;i++)
        for(j=1;j<=2*n;j++)
            if(j==(i+n))
                a[i][j]=1;

    for(i=n;i>1;i--)
    {
        if(a[i-1][1]<a[i][1])
        for(j=1;j<=n*2;j++)
        {
            d=a[i][j];
            a[i][j]=a[i-1][j];
            a[i-1][j]=d;
        }
    }
    cout<<"Augmented : "<<endl;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n*2;j++)
            cout<<a[i][j]<<"    ";
        cout<<endl;
    }

    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n*2;j++)
        if(j!=i)
        {
            d=a[j][i]/a[i][i];
            for(k=1;k<=n*2;k++)
                a[j][k]-=a[i][k]*d;
        }
    }

    for(i=1;i<=n;i++)
    {
    d=a[i][i];
        for(j=1;j<=n*2;j++)
            a[i][j]=a[i][j]/d;
    }


    cout<<"Inverse Matrix : "<<endl;
    for(i=1;i<=n;i++)
    {
        for(j=n+1;j<=n*2;j++)
            cout<<a[i][j]<<"    ";
        cout<<endl;
    }

    getch();
}
